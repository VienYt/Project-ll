import tkinter as tk from tkinter import messagebox from PIL import Image, ImageTk import requests from io import BytesIO def display_cat_photo(): url = "https://cataas.com/cat" response = requests.get(url) img_data = response.content img = Image.open(BytesIO(img_data)) img = img.resize((300, 200), Image.ANTIALIAS) photo = ImageTk.PhotoImage(img) label.config(image=photo) label.image = photo def display_hearts(): messagebox.showinfo("❤️", "❤️ ❤️ ❤️") def ask_to_be_gf(): response = messagebox.askquestion("Wanna be my gf", "❤️") if response == 'yes': display_hearts() else: display_cat_photo() root = tk.Tk() root.title("Love") label = tk.Label(root) label.pack() yes_button = tk.Button(root, text="Yes", command=lambda: ask_to_be_gf()) yes_button.pack(side=tk.LEFT, padx=10) no_button = tk.Button(root, text="No", command=lambda: display_cat_photo()) no_button.pack(side=tk.RIGHT, padx=10) display_cat_photo() root.mainloop()
